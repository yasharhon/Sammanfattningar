\documentclass[tikz]{standalone}
\usepackage{pgfplots}

\pgfplotsset{
	compat = 1.8
}

\begin{document}

\begin{tikzpicture}
\begin{axis}[
    view = {145}{10},
	axis equal,
	axis lines = center,
    xlabel={$x$},ylabel={$y$},zlabel={$z$},
    ticks = none,
    colormap/cool,
    set layers
]
\addplot3[
	surf,
    domain=0:2*pi,y domain=0:2,
    z buffer=sort,
    on layer = axis background
    ]
    ({sinh(y)*cos(deg(x))}, {1.5*sinh(y)*sin(deg(x))}, {cosh(y)});
]
\addplot3[
	surf,
    domain=0:2*pi,y domain=0:2,
    z buffer=sort,
    on layer = axis background
    ]
    ({sinh(y)*cos(deg(x))}, {1.5*sinh(y)*sin(deg(x))}, {-cosh(y)});
]
\end{axis}
\end{tikzpicture}

\end{document}
